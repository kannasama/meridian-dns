## Key Reference: Existing Patterns

### Route handler pattern
```cpp
CROW_ROUTE(app, "/api/v1/zones/<int>/clone")
  .methods("POST"_method)(
    [this](const crow::request& req, int iZoneId) -> crow::response {
      try {
        auto rcCtx = rh::authenticate(_amMiddleware, req);
        rh::requirePermission(rcCtx, Permissions::kZonesEdit);
        rh::enforceBodyLimit(req);
        auto jBody = nlohmann::json::parse(req.body);
        // ... logic ...
        return rh::jsonResponse(201, result);
      } catch (const common::AppError& e) { return rh::errorResponse(e); }
        catch (const nlohmann::json::exception&) { return rh::invalidJsonResponse(); }
    });
```
Note: `authenticate`, `requirePermission`, `enforceBodyLimit`, `jsonResponse`, `errorResponse`,
`invalidJsonResponse` are all free functions in `namespace dns::api` from
`include/api/RouteHelpers.hpp`. They are NOT methods; call them unqualified after
`using namespace dns::api;` or fully qualified as `dns::api::authenticate(...)` etc.

### Repository connection pool pattern
```cpp
auto cg = _cpPool.checkout();
pqxx::work txn(*cg);
auto result = txn.exec("SELECT ...", pqxx::params{...});
txn.commit();
```

```cpp
#include <chrono>
#include <cstdint>
#include <optional>
#include <string>
#include <vector>

namespace dns::dal {

class ConnectionPool;

/// Row type returned from tag queries.
struct TagRow {
  int64_t iId = 0;
  std::string sName;
  int iZoneCount = 0;  // populated by listWithCounts()
  std::chrono::system_clock::time_point tpCreatedAt;
};

/// Manages the tags vocabulary table.
/// Class abbreviation: tr
class TagRepository {
 public:
  explicit TagRepository(ConnectionPool& cpPool);
  ~TagRepository();

  /// List all tags with per-tag zone count (LEFT JOIN zones on tags.name = ANY(zones.tags)).
  std::vector<TagRow> listWithCounts();

  /// Find a tag by ID. Returns nullopt if not found.
  std::optional<TagRow> findById(int64_t iId);

  /// Rename a tag. Cascades: array_replace on zones.tags to update all zone tag arrays.
  /// Returns the updated row. Throws NotFoundError if tag does not exist.
  TagRow rename(int64_t iId, const std::string& sNewName);

  /// Delete a tag. Cascades: array_remove on zones.tags to remove from all zone tag arrays.
  /// Throws NotFoundError if tag does not exist.
  void deleteTag(int64_t iId);

  /// Upsert tags into vocabulary by name. For each name not already present, inserts a new
  /// row. Uses INSERT ... ON CONFLICT (name) DO NOTHING. Returns the number inserted.
  int upsertVocabulary(const std::vector<std::string>& vTagNames);

 private:
  ConnectionPool& _cpPool;
};

}  // namespace dns::dal
```

### Tag1.2 — Write the implementation

Create `src/dal/TagRepository.cpp`:

- `listWithCounts()` SQL:
  ```sql
  SELECT t.id, t.name, COUNT(z.id)::int AS zone_count, t.created_at
  FROM tags t
  LEFT JOIN zones z ON t.name = ANY(z.tags)
  GROUP BY t.id, t.name, t.created_at
  ORDER BY t.name
  ```
  Map each row to `TagRow{iId, sName, iZoneCount, tpCreatedAt}`.

- `findById(iId)` SQL:
  ```sql
  SELECT t.id, t.name, COUNT(z.id)::int AS zone_count, t.created_at
  FROM tags t
  LEFT JOIN zones z ON t.name = ANY(z.tags)
  WHERE t.id = $1
  GROUP BY t.id, t.name, t.created_at
  ```
  Returns `std::nullopt` if no rows.

- `rename(iId, sNewName)`:
  - Execute in a single transaction:
    1. Fetch old name: `SELECT name FROM tags WHERE id = $1 FOR UPDATE`
    2. If no row, throw `common::NotFoundError("TAG_NOT_FOUND", "Tag not found")`
    3. Update vocabulary: `UPDATE tags SET name = $2 WHERE id = $1`
    4. Cascade to zones:
       ```sql
       UPDATE zones SET tags = array_replace(tags, $1::text, $2::text)
       WHERE $1::text = ANY(tags)
       ```
       (Pass old name as `$1`, new name as `$2`.)
  - Return `findById(iId)` result after commit.

- `deleteTag(iId)`:
  - Execute in a single transaction:
    1. Fetch name: `SELECT name FROM tags WHERE id = $1 FOR UPDATE`
    2. If no row, throw `NotFoundError`
    3. Cascade to zones:
       ```sql
       UPDATE zones SET tags = array_remove(tags, $1::text)
       WHERE $1::text = ANY(tags)
       ```
    4. Delete vocabulary: `DELETE FROM tags WHERE id = $2` (use the `iId`)
  - Commit.

- `upsertVocabulary(vTagNames)`:
  - If `vTagNames` is empty, return 0.
  - Use `pqxx::work`, loop over names and execute:
    ```sql
    INSERT INTO tags (name, created_at)
    VALUES ($1, NOW())
    ON CONFLICT (name) DO NOTHING
    ```
  - `txn.commit()`. Return count of names that did not conflict (track by checking `cmd_tuples()` on each exec).

### Task2.2 — Implement cloneZone()

In `src/dal/ZoneRepository.cpp`, implement `cloneZone()` as a single atomic transaction:

```
1. SELECT id, name, manage_soa, manage_ns, deployment_retention, git_repo_id, git_branch
   FROM zones WHERE id = $1
   → If no row, throw NotFoundError("ZONE_NOT_FOUND", "Source zone not found")

2. INSERT INTO zones (name, view_id, deployment_retention, manage_soa, manage_ns,
       template_id, soa_preset_id, tags, created_at)
   VALUES ($newName, $iViewId, $retention, $bManageSoa, $bManageNs,
           NULL, NULL, '{}', NOW())
   RETURNING id
   → iNewZoneId

3. INSERT INTO records (zone_id, name, type, ttl, value_template, priority,
       provider_meta, pending_delete, created_at, updated_at)
   SELECT $iNewZoneId, name, type, ttl, value_template, priority,
          provider_meta, false, NOW(), NOW()
   FROM records
   WHERE zone_id = $iSourceId AND NOT pending_delete

4. txn.commit()
5. Return iNewZoneId
```

Note: git_repo_id and git_branch are intentionally NOT copied (the clone starts fresh with no
GitOps linkage).

### Task3.3 — Implement search()

In `src/dal/RecordRepository.cpp`, implement `search()`:

Base query:
```sql
SELECT r.id, r.zone_id, z.name AS zone_name, vi.name AS view_name,
       r.name, r.type, r.ttl, r.value_template, r.priority
FROM records r
JOIN zones z ON r.zone_id = z.id
JOIN views vi ON z.view_id = vi.id
WHERE NOT r.pending_delete
  AND (r.name ILIKE '%' || $1 || '%'
       OR r.value_template ILIKE '%' || $1 || '%'
       OR z.name ILIKE '%' || $1 || '%')
```

Build the WHERE clause dynamically using `pqxx::params`:
- Param `$1` is always the query string.
- If `osType` is set, append `AND r.type = $N` (track parameter index).
- If `oiZoneId` is set, append `AND r.zone_id = $N`.
- If `oiViewId` is set, append `AND z.view_id = $N`.
- Append `ORDER BY z.name, r.name, r.type LIMIT 200`.

Map results to `SearchResult` structs. Return the vector.

Implementation note: use `pqxx::work txn(*cg)` (read-only is fine; no commit needed for
SELECT, but committing an empty transaction is harmless).

```cpp
          if (auto pViewId = req.url_params.get("view_id")) {
            oiViewId = std::stoll(pViewId);

            jArr.push_back({
              {"id", sr.iId},
              {"zone_id", sr.iZoneId},
              {"zone_name", sr.sZoneName},
              {"view_name", sr.sViewName},
              {"name", sr.sName},
              {"type", sr.sType},
              {"ttl", sr.iTtl},
              {"value_template", sr.sValueTemplate},
              {"priority", sr.iPriority},
            });
          }

        } catch (const common::AppError& e) { return errorResponse(e); }
      });
}
```

### Task16.1 — Create SearchView.vue

Create `ui/src/views/SearchView.vue`:

```vue
<script setup lang="ts">
import { ref, watch, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { searchRecords } from '../api/search'
import type { SearchResult } from '../types'
import { useNotification } from '../stores/notification'

const route = useRoute()
const router = useRouter()
const notify = useNotification()

const results = ref<SearchResult[]>([])
const loading = ref(false)
const query = ref('')

async function performSearch(q: string) {
  if (!q.trim()) {
    results.value = []
    return
  }
  loading.value = true
  try {
    results.value = await searchRecords({ q })
  } catch {
    notify.error('Search failed')
  } finally {
    loading.value = false
  }
}

function goToZone(zoneId: number) {
  router.push(`/zones/${zoneId}`)
}

// React to URL query param changes
watch(
  () => route.query.q as string | undefined,
  (newQ) => {
    query.value = newQ ?? ''
    if (newQ) performSearch(newQ)
  },
  { immediate: true },
)
</script>

<template>
  <div class="p-4">
    <div class="flex justify-between items-center mb-4">
      <h1 class="text-xl font-semibold">
        Record Search
        <span v-if="query" class="text-muted font-normal text-base ml-2">
          — "{{ query }}"
        </span>
      </h1>
    </div>

    <DataTable :value="results" :loading="loading" class="text-sm"
               @rowClick="e => goToZone(e.data.zone_id)">
      <Column field="name" header="Name" />
      <Column field="type" header="Type" />
      <Column field="ttl" header="TTL" />
      <Column field="value_template" header="Value" />
      <Column header="Zone">
        <template #body="{ data }">
          <router-link :to="`/zones/${data.zone_id}`" class="text-primary">
            {{ data.zone_name }}
          </router-link>
        </template>
      </Column>
      <Column field="view_name" header="View" />
    </DataTable>

    <p v-if="!loading && results.length === 0 && query"
       class="text-muted text-sm mt-4">
      No records found for "{{ query }}"
    </p>
  </div>
</template>
```
