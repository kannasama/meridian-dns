```sql
-- SPDX-License-Identifier: AGPL-3.0-or-later
-- v016: Templates & Snippets (Group 1)

-- SOA presets: named SOA timing configurations, support {{var}} in mname/rname
CREATE TABLE soa_presets (
  id             BIGSERIAL PRIMARY KEY,
  name           TEXT NOT NULL UNIQUE,
  mname_template TEXT NOT NULL,
  rname_template TEXT NOT NULL,
  refresh        INTEGER NOT NULL DEFAULT 3600,
  retry          INTEGER NOT NULL DEFAULT 900,
  expire         INTEGER NOT NULL DEFAULT 604800,
  minimum        INTEGER NOT NULL DEFAULT 300,
  default_ttl    INTEGER NOT NULL DEFAULT 3600,
  created_at     TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at     TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Snippets: named reusable record sets
CREATE TABLE snippets (
  id          BIGSERIAL PRIMARY KEY,
  name        TEXT NOT NULL UNIQUE,
  description TEXT NOT NULL DEFAULT '',
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Records belonging to a snippet (no provider_meta — snippets define DNS structure only)
CREATE TABLE snippet_records (
  id             BIGSERIAL PRIMARY KEY,
  snippet_id     BIGINT NOT NULL REFERENCES snippets(id) ON DELETE CASCADE,
  name           TEXT NOT NULL,
  type           TEXT NOT NULL,
  ttl            INTEGER NOT NULL DEFAULT 300,
  value_template TEXT NOT NULL,
  priority       INTEGER NOT NULL DEFAULT 0,
  sort_order     INTEGER NOT NULL DEFAULT 0,
  created_at     TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Zone templates: ordered collection of snippets + optional SOA preset
CREATE TABLE zone_templates (
  id            BIGSERIAL PRIMARY KEY,
  name          TEXT NOT NULL UNIQUE,
  description   TEXT NOT NULL DEFAULT '',
  soa_preset_id BIGINT REFERENCES soa_presets(id) ON DELETE SET NULL,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Ordered snippet membership in a template
CREATE TABLE zone_template_snippets (
  template_id BIGINT NOT NULL REFERENCES zone_templates(id) ON DELETE CASCADE,
  snippet_id  BIGINT NOT NULL REFERENCES snippets(id) ON DELETE RESTRICT,
  sort_order  INTEGER NOT NULL DEFAULT 0,
  PRIMARY KEY (template_id, snippet_id)
);

-- Extend zones with template link, compliance flag, SOA preset, and tags
ALTER TABLE zones
  ADD COLUMN template_id            BIGINT REFERENCES zone_templates(id) ON DELETE SET NULL,
  ADD COLUMN template_check_pending BOOLEAN NOT NULL DEFAULT FALSE,
  ADD COLUMN soa_preset_id          BIGINT REFERENCES soa_presets(id) ON DELETE SET NULL,
  ADD COLUMN tags                   TEXT[] NOT NULL DEFAULT '{}';

CREATE INDEX idx_zones_template_id ON zones(template_id) WHERE template_id IS NOT NULL;
CREATE INDEX idx_zones_tags ON zones USING GIN(tags);
CREATE INDEX idx_snippet_records_snippet_id ON snippet_records(snippet_id);
CREATE INDEX idx_zone_template_snippets_template ON zone_template_snippets(template_id, sort_order);
```

```cpp
// src/dal/SnippetRepository.cpp
// SPDX-License-Identifier: AGPL-3.0-or-later
// Copyright (c) 2026 Meridian DNS Contributors
#include "dal/SnippetRepository.hpp"
#include "common/Errors.hpp"
#include "dal/ConnectionPool.hpp"
#include <pqxx/pqxx>

namespace dns::dal {

SnippetRepository::SnippetRepository(ConnectionPool& cpPool) : _cpPool(cpPool) {}
SnippetRepository::~SnippetRepository() = default;

int64_t SnippetRepository::create(const std::string& sName, const std::string& sDescription) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  try {
    auto r = txn.exec(
        "INSERT INTO snippets (name, description) VALUES ($1, $2) RETURNING id",
        pqxx::params{sName, sDescription});
    txn.commit();
    return r.one_row()[0].as<int64_t>();
  } catch (const pqxx::unique_violation&) {
    throw common::ConflictError("SNIPPET_EXISTS", "Snippet '" + sName + "' already exists");
  }
}

std::vector<SnippetRow> SnippetRepository::listAll() {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto r = txn.exec(
      "SELECT id, name, description, "
      "EXTRACT(EPOCH FROM created_at)::bigint, "
      "EXTRACT(EPOCH FROM updated_at)::bigint "
      "FROM snippets ORDER BY name");
  txn.commit();
  std::vector<SnippetRow> vRows;
  vRows.reserve(r.size());
  for (const auto& row : r) {
    SnippetRow sr;
    sr.iId          = row[0].as<int64_t>();
    sr.sName        = row[1].as<std::string>();
    sr.sDescription = row[2].as<std::string>();
    sr.tpCreatedAt  = std::chrono::system_clock::time_point(
        std::chrono::seconds(row[3].as<int64_t>()));
    sr.tpUpdatedAt  = std::chrono::system_clock::time_point(
        std::chrono::seconds(row[4].as<int64_t>()));
    vRows.push_back(std::move(sr));
  }
  return vRows;
}

std::optional<SnippetRow> SnippetRepository::findById(int64_t iId) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto r = txn.exec(
      "SELECT id, name, description, "
      "EXTRACT(EPOCH FROM created_at)::bigint, "
      "EXTRACT(EPOCH FROM updated_at)::bigint "
      "FROM snippets WHERE id = $1",
      pqxx::params{iId});
  if (r.empty()) { txn.commit(); return std::nullopt; }
  SnippetRow sr;
  sr.iId          = r[0][0].as<int64_t>();
  sr.sName        = r[0][1].as<std::string>();
  sr.sDescription = r[0][2].as<std::string>();
  sr.tpCreatedAt  = std::chrono::system_clock::time_point(
      std::chrono::seconds(r[0][3].as<int64_t>()));
  sr.tpUpdatedAt  = std::chrono::system_clock::time_point(
      std::chrono::seconds(r[0][4].as<int64_t>()));
  sr.vRecords = listRecordsInternal(txn, iId);
  txn.commit();
  return sr;
}

// Private helper used by findById (within existing txn)
// Declared inline here; add to private section in .hpp if needed
static std::vector<SnippetRecordRow> listRecordsInternal(pqxx::work& txn, int64_t iSnippetId) {
  auto r = txn.exec(
      "SELECT id, snippet_id, name, type, ttl, value_template, priority, sort_order "
      "FROM snippet_records WHERE snippet_id = $1 ORDER BY sort_order, id",
      pqxx::params{iSnippetId});
  std::vector<SnippetRecordRow> vRows;
  vRows.reserve(r.size());
  for (const auto& row : r) {
    SnippetRecordRow srr;
    srr.iId            = row[0].as<int64_t>();
    srr.iSnippetId     = row[1].as<int64_t>();
    srr.sName          = row[2].as<std::string>();
    srr.sType          = row[3].as<std::string>();
    srr.iTtl           = row[4].as<int>();
    srr.sValueTemplate = row[5].as<std::string>();
    srr.iPriority      = row[6].as<int>();
    srr.iSortOrder     = row[7].as<int>();
    vRows.push_back(std::move(srr));
  }
  return vRows;
}

std::vector<SnippetRecordRow> SnippetRepository::listRecords(int64_t iSnippetId) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  auto vRows = listRecordsInternal(txn, iSnippetId);
  txn.commit();
  return vRows;
}

void SnippetRepository::update(int64_t iId, const std::string& sName,
                               const std::string& sDescription) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  try {
    txn.exec(
        "UPDATE snippets SET name=$1, description=$2, updated_at=NOW() WHERE id=$3",
        pqxx::params{sName, sDescription, iId});
    txn.commit();
  } catch (const pqxx::unique_violation&) {
    throw common::ConflictError("SNIPPET_EXISTS", "Snippet '" + sName + "' already exists");
  }
}

void SnippetRepository::deleteById(int64_t iId) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  try {
    txn.exec("DELETE FROM snippets WHERE id=$1", pqxx::params{iId});
    txn.commit();
  } catch (const pqxx::foreign_key_violation&) {
    throw common::ConflictError("SNIPPET_IN_USE",
        "Snippet is referenced by one or more templates");
  }
}

void SnippetRepository::replaceRecords(int64_t iSnippetId,
                                       const std::vector<SnippetRecordRow>& vRecords) {
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  txn.exec("DELETE FROM snippet_records WHERE snippet_id=$1", pqxx::params{iSnippetId});
  for (const auto& r : vRecords) {
    txn.exec(
        "INSERT INTO snippet_records (snippet_id, name, type, ttl, value_template, priority, sort_order) "
        "VALUES ($1,$2,$3,$4,$5,$6,$7)",
        pqxx::params{iSnippetId, r.sName, r.sType, r.iTtl, r.sValueTemplate,
                     r.iPriority, r.iSortOrder});
  }
  txn.commit();
}

}  // namespace dns::dal
```

```cpp
  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);
  // Build ARRAY['tag1','tag2'] or ARRAY[]::TEXT[]
  std::string sArr = "ARRAY[";
  for (size_t i = 0; i < vTags.size(); ++i) {
```

```cpp
// src/core/TemplateEngine.cpp
// SPDX-License-Identifier: AGPL-3.0-or-later
// Copyright (c) 2026 Meridian DNS Contributors
#include "core/TemplateEngine.hpp"
#include <chrono>
#include <unordered_map>

namespace dns::core {

common::PreviewResult TemplateEngine::computeComplianceDiff(
    int64_t iZoneId,
    const std::string& sZoneName,
    const std::vector<dal::SnippetRecordRow>& vExpected,
    const std::vector<dal::RecordRow>& vCurrent) {

  // Build lookup: (name, type) -> current zone record
  std::unordered_map<std::string, const dal::RecordRow*> mCurrent;
  for (const auto& r : vCurrent) {
    mCurrent[r.sName + "\0" + r.sType] = &r;
  }

  common::PreviewResult result;
  result.iZoneId  = iZoneId;
  result.sZoneName = sZoneName;
  result.tpGeneratedAt = std::chrono::system_clock::now();

  for (const auto& expected : vExpected) {
    const std::string sKey = expected.sName + "\0" + expected.sType;
    auto it = mCurrent.find(sKey);

    if (it == mCurrent.end()) {
      // Missing — Add
      common::RecordDiff diff;
      diff.action       = common::DiffAction::Add;
      diff.sName        = expected.sName;
      diff.sType        = expected.sType;
      diff.sSourceValue = expected.sValueTemplate;
      diff.uTtl         = static_cast<uint32_t>(expected.iTtl);
      diff.iPriority    = expected.iPriority;
      result.vDiffs.push_back(std::move(diff));
    } else {
      const auto* pCurrent = it->second;
      bool bValueDiffers = (pCurrent->sValueTemplate != expected.sValueTemplate);
      bool bTtlDiffers   = (pCurrent->iTtl != expected.iTtl);
      bool bPriDiffers   = (pCurrent->iPriority != expected.iPriority);

      if (bValueDiffers || bTtlDiffers || bPriDiffers) {
        // Exists but different — Update
        common::RecordDiff diff;
        diff.action        = common::DiffAction::Update;
        diff.sName         = expected.sName;
        diff.sType         = expected.sType;
        diff.sSourceValue  = expected.sValueTemplate;
        diff.sProviderValue = pCurrent->sValueTemplate;
        diff.uTtl          = static_cast<uint32_t>(expected.iTtl);
        diff.iPriority     = expected.iPriority;
        result.vDiffs.push_back(std::move(diff));
      }
      // Identical — no diff
    }
  }

  return result;
}

}  // namespace dns::core
```

| PUT | `/api/v1/templates/<int>` | `kTemplatesEdit` | `update()` + `setSnippets()` (flags linked zones) |
| DELETE | `/api/v1/templates/<int>` | `kTemplatesDelete` | `deleteById()` |
| POST | `/api/v1/zones/<int>/template/push` | `kZonesDeploy` | see below |
| GET | `/api/v1/zones/<int>/template/check` | `kZonesView` | see below |
| POST | `/api/v1/zones/<int>/template/apply` | `kRecordsCreate` | see below |
| DELETE | `/api/v1/zones/<int>/template` | `kZonesEdit` | `setTemplateLink(zoneId, nullopt)` + `clearTemplateCheckPending()` |

**Push template logic** (`POST /api/v1/zones/<int>/template/push`):

```
Body: { "template_id": N, "link": true }

1. findById(template_id) → 404 if not found
2. Get all expected records (iterate template.vSnippetIds → snrRepo.listRecords each)
3. If link=true:
   - zrRepo.setTemplateLink(zoneId, templateId)   // sets template_id + pending=true
   - Run compliance check → return PreviewResult JSON
4. If link=false (one-shot apply):
   - For each expected record: rrRepo.create(...)
   - Return { "message": "Template applied", "record_ids": [...] }
```

**Compliance check logic** (`GET /api/v1/zones/<int>/template/check`):

```
1. zrRepo.findById(zoneId) → 404 if not found
2. If zone.oTemplateId is null → 400 "ZONE_NOT_LINKED"
3. ztrRepo.findById(*zone.oTemplateId) → 404 if not found
4. Build expected records (all snippet records in template order)
5. rrRepo.listByZoneId(zoneId) → vCurrentRecords
6. TemplateEngine::computeComplianceDiff(zoneId, zone.sName, vExpected, vCurrent)
7. Return PreviewResult as JSON (same shape as /zones/{id}/preview)
```

**Apply compliance records** (`POST /api/v1/zones/<int>/template/apply`):

```
Body: { "records": [{ "name": "www", "type": "CNAME" }] }

1. Re-run compliance check (same as GET /check)
2. Build a set of requested name+type pairs from body
3. For each diff in result.vDiffs:
   - If diff.action == Add AND name+type in requested set:
     rrRepo.create(zoneId, diff.sName, diff.sType, diff.uTtl, diff.sSourceValue, diff.iPriority)
   - If diff.action == Update AND name+type in requested set:
     find existing record by name+type → rrRepo.update(...)
4. zrRepo.clearTemplateCheckPending(zoneId)
5. Return { "message": "Compliance records applied" }
```

```vue
import { ref, onMounted, computed } from 'vue'
import { useNotification } from '../stores/notification'

import * as snippetApi from '../api/snippets'

  try {
    snippets.value = await snippetApi.listSnippets()
  } finally {
    loading.value = false
  }

  editingId.value = snippet.id
  const full = await snippetApi.getSnippet(snippet.id)
  form.value = { name: full.name, description: full.description, records: [...full.records] }
  drawerVisible.value = true
}

function addRecord() {
  form.value.records.push({
    name: '', type: 'A', ttl: 300, value_template: '', priority: 0,
    sort_order: form.value.records.length
  })
}

function removeRecord(idx: number) {
  form.value.records.splice(idx, 1)
  form.value.records.forEach((r, i) => { r.sort_order = i })
}

async function save() {
  try {
    if (editingId.value === null) {
      await snippetApi.createSnippet(form.value)
      addToast('Snippet created', 'success')
    } else {
      await snippetApi.updateSnippet(editingId.value, form.value)
      addToast('Snippet updated', 'success')
    }
    drawerVisible.value = false
    await fetchSnippets()
  } catch (e: any) {
    addToast(e.message ?? 'Save failed', 'error')
  }
}

async function remove(snippet: Snippet) {
  if (!await confirmDelete(snippet.name)) return
  try {
    await snippetApi.deleteSnippet(snippet.id)
    addToast('Snippet deleted', 'success')
    await fetchSnippets()
  } catch (e: any) {
    addToast(e.message ?? 'Delete failed', 'error')
  }
}

onMounted(fetchSnippets)
</script>
```
