# v1.1 Group 2: Dynamic Variables — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add built-in `{{sys.*}}` variables and user-defined dynamic variables (strftime-based) to the variable expansion pipeline, with schema persistence for the serial counter and a two-tab UI.

**Architecture:** The `VariableEngine::expand()` method gains an optional `SysContext` parameter carrying the zone name and pre-fetched serial. Built-in sys vars (date, datetime, timestamp, zone, serial) are resolved inside `expand()`. User-defined dynamic variables are stored in the `variables` table with `variable_kind='dynamic'` and a `dynamic_format` strftime string; they are resolved at expand time. `SystemConfigRepository` manages the durable serial counter in the existing `system_config` table. `DeploymentEngine` obtains the serial before pushing and passes it via `SysContext`.

**Tech Stack:** C++20, libpqxx, `std::strftime` (UTC via `gmtime_r`), nlohmann/json, Vue 3 + TypeScript + PrimeVue. Build: `docker buildx build --load -t meridian-dns .` in the worktree.

**Step 3: Implement getAndIncrementSerial()**

```cpp
// SPDX-License-Identifier: AGPL-3.0-or-later
// Copyright (c) 2026 Meridian DNS Contributors
// This file is part of Meridian DNS. See LICENSE for details.

#include "dal/SystemConfigRepository.hpp"

#include <chrono>
#include <ctime>
#include <iomanip>
#include <sstream>
#include <stdexcept>

#include "dal/ConnectionPool.hpp"

namespace dns::dal {

SystemConfigRepository::SystemConfigRepository(ConnectionPool& cpPool)
    : _cpPool(cpPool) {}

SystemConfigRepository::~SystemConfigRepository() = default;

std::string SystemConfigRepository::getAndIncrementSerial() {
  // Compute today's UTC date as YYYYMMDD
  auto now = std::chrono::system_clock::now();
  auto tt = std::chrono::system_clock::to_time_t(now);
  std::tm tmUtc{};
  gmtime_r(&tt, &tmUtc);
  char sDateBuf[9];
  std::strftime(sDateBuf, sizeof(sDateBuf), "%Y%m%d", &tmUtc);
  std::string sTodayDate(sDateBuf);

  auto cg = _cpPool.checkout();
  pqxx::work txn(*cg);

  // Lock both rows for update
  auto result = txn.exec(
      "SELECT key, value FROM system_config "
      "WHERE key IN ('serial_counter_date', 'serial_counter_seq') "
      "ORDER BY key FOR UPDATE");

  std::string sStoredDate;
  int iSeq = 0;
  for (const auto& row : result) {
    std::string sKey = row["key"].as<std::string>();
    if (sKey == "serial_counter_date") sStoredDate = row["value"].as<std::string>();
    if (sKey == "serial_counter_seq")  iSeq = row["value"].as<int>();
  }

  // Reset on new day; increment on same day
  if (sStoredDate != sTodayDate) {
    sStoredDate = sTodayDate;
    iSeq = 0;
  } else {
    iSeq++;
  }
  if (iSeq > 99) {
    throw std::runtime_error("SOA serial suffix exhausted for today (max 99)");
  }

  txn.exec("UPDATE system_config SET value=$1, updated_at=NOW() WHERE key='serial_counter_date'",
           pqxx::params{sTodayDate});
  txn.exec("UPDATE system_config SET value=$1, updated_at=NOW() WHERE key='serial_counter_seq'",
           pqxx::params{std::to_string(iSeq)});
  txn.commit();

  // Format: YYYYMMDDNN (10 chars, NN zero-padded)
  std::ostringstream oss;
  oss << sTodayDate << std::setfill('0') << std::setw(2) << iSeq;
  return oss.str();
}

}  // namespace dns::dal
```

```cpp
std::string VariableEngine::expand(const std::string& sTmpl, int64_t iZoneId,
                                    const SysContext& sysCtx) const {
  if (!_pVarRepo) {
    throw std::runtime_error("VariableEngine: no VariableRepository injected");
  }

  auto vDeps = listDependencies(sTmpl);
  if (vDeps.empty()) return sTmpl;

  // Fetch static + dynamic variables for this zone
  auto vRows = _pVarRepo->listByZoneId(iZoneId);

  // Build lookup maps: static (zone-scoped > global) and dynamic by name
  std::unordered_map<std::string, std::string> mStatic;
  std::unordered_map<std::string, std::string> mDynamicFormat;
  for (const auto& row : vRows) {
    if (row.sVariableKind == "dynamic") {
      if (row.osDynamicFormat.has_value()) {
        mDynamicFormat[row.sName] = *row.osDynamicFormat;
      }
    } else {
      if (row.sScope == "global") mStatic.emplace(row.sName, row.sValue);
    }
  }
  for (const auto& row : vRows) {
    if (row.sVariableKind == "static" && row.sScope == "zone") {
      mStatic[row.sName] = row.sValue;  // overwrites global
    }
  }

  // Capture current UTC time once for all dynamic resolutions
  auto now = std::chrono::system_clock::now();
  auto tt = std::chrono::system_clock::to_time_t(now);
  std::tm tmUtc{};
  gmtime_r(&tt, &tmUtc);

  static const std::regex reVar(R"(\{\{([A-Za-z0-9_.]+)\}\})");
  std::string sResult;
  auto itBegin = std::sregex_iterator(sTmpl.begin(), sTmpl.end(), reVar);
  auto itEnd   = std::sregex_iterator();

  size_t iLastPos = 0;
  for (auto it = itBegin; it != itEnd; ++it) {
    sResult.append(sTmpl, iLastPos, static_cast<size_t>(it->position()) - iLastPos);
    std::string sVarName = (*it)[1].str();

    std::string sResolved;
    if (sVarName == "sys.zone") {
      if (sysCtx.sZoneName.empty()) throw common::UnresolvedVariableError(
          "UNDEFINED_VARIABLE", "{{sys.zone}} requires a zone context");
      sResolved = sysCtx.sZoneName;
    } else if (sVarName == "sys.serial") {
      if (sysCtx.sSerial.empty()) throw common::UnresolvedVariableError(
          "UNDEFINED_VARIABLE", "{{sys.serial}} is only available during deployment push");
      sResolved = sysCtx.sSerial;
    } else if (sVarName == "sys.date") {
      char buf[9];
      std::strftime(buf, sizeof(buf), "%Y%m%d", &tmUtc);
      sResolved = buf;
    } else if (sVarName == "sys.datetime") {
      char buf[15];
      std::strftime(buf, sizeof(buf), "%Y%m%d%H%M%S", &tmUtc);
      sResolved = buf;
    } else if (sVarName == "sys.timestamp") {
      sResolved = std::to_string(static_cast<int64_t>(tt));
    } else {
      // Check static variables first
      auto itStatic = mStatic.find(sVarName);
      if (itStatic != mStatic.end()) {
        sResolved = itStatic->second;
      } else {
        // Check user-defined dynamic variables
        auto itDyn = mDynamicFormat.find(sVarName);
        if (itDyn != mDynamicFormat.end()) {
          char buf[128] = {};
          std::strftime(buf, sizeof(buf), itDyn->second.c_str(), &tmUtc);
          sResolved = buf;
        } else {
          throw common::UnresolvedVariableError(
              "UNDEFINED_VARIABLE",
              "Variable '" + sVarName + "' not found for zone " + std::to_string(iZoneId));
        }
      }
    }

    sResult.append(sResolved);
    iLastPos = static_cast<size_t>(it->position() + it->length());
  }
  sResult.append(sTmpl, iLastPos);
  return sResult;
}
```

Update `validate()` to treat `sys.*` variables and dynamic variables as resolvable:

```cpp
bool VariableEngine::validate(const std::string& sTmpl, int64_t iZoneId) const {
  if (!_pVarRepo) {
    throw std::runtime_error("VariableEngine: no VariableRepository injected");
  }

  auto vDeps = listDependencies(sTmpl);
  if (vDeps.empty()) return true;

  auto vRows = _pVarRepo->listByZoneId(iZoneId);
  std::unordered_map<std::string, bool> mExists;
  for (const auto& row : vRows) mExists[row.sName] = true;

  static const std::string sSysPrefix = "sys.";
  for (const auto& sDep : vDeps) {
    // sys.* variables are always resolvable (built-in)
    if (sDep.rfind(sSysPrefix, 0) == 0) continue;
    if (mExists.find(sDep) == mExists.end()) return false;
  }
  return true;
}
```

```typescript
import { BUILTIN_SYS_VARS } from '../api/variables'
import type { BuiltinSysVar } from '../api/variables'

const builtinSysVars = BUILTIN_SYS_VARS  // static, no loading needed
const dynamicVariables = ref<Variable[]>([])
const loadingDynamic = ref(false)
const dynDrawerVisible = ref(false)
const editingDynId = ref<number | null>(null)
const dynForm = ref({ name: '', dynamic_format: '' })

async function fetchDynamic() {
  loadingDynamic.value = true
  try {
    const all = await varApi.listVariables()
    dynamicVariables.value = all.filter(v => v.variable_kind === 'dynamic')
  } catch {
    notify.error('Failed to load dynamic variables')
  } finally {
    loadingDynamic.value = false
  }
}

function openCreateDynamic() {
  editingDynId.value = null
  dynForm.value = { name: '', dynamic_format: '' }
  dynDrawerVisible.value = true
}

function openEditDynamic(v: Variable) {
  editingDynId.value = v.id
  dynForm.value = { name: v.name, dynamic_format: v.dynamic_format ?? '' }
  dynDrawerVisible.value = true
}

async function saveDynamic() {
  try {
    if (editingDynId.value === null) {
      await varApi.createVariable({
        name: dynForm.value.name,
        value: '',
        type: 'string',
        scope: 'global',
        variable_kind: 'dynamic',
        dynamic_format: dynForm.value.dynamic_format,
      })
      notify.success('Dynamic variable created')
    } else {
      await varApi.updateVariableFull(
        editingDynId.value, '', dynForm.value.dynamic_format
      )
      notify.success('Dynamic variable updated')
    }
    dynDrawerVisible.value = false
    await fetchDynamic()
  } catch (e: unknown) {
    notify.error(e instanceof Error ? e.message : 'Save failed')
  }
}

async function handleDeleteDynamic(v: Variable) {
  if (!await confirmDelete(v.name)) return
  try {
    await varApi.deleteVariable(v.id)
    notify.success('Dynamic variable deleted')
    await fetchDynamic()
  } catch (e: unknown) {
    notify.error(e instanceof Error ? e.message : 'Delete failed')
  }
}

function previewDynamic(format: string | null): string {
  if (!format) return '—'
  // Simple client-side preview using Intl or manual strftime mapping
  // For common patterns: %Y %m %d %H %M %S
  const d = new Date()
  return format
    .replace('%Y', String(d.getUTCFullYear()))
    .replace('%m', String(d.getUTCMonth() + 1).padStart(2, '0'))
    .replace('%d', String(d.getUTCDate()).padStart(2, '0'))
    .replace('%H', String(d.getUTCHours()).padStart(2, '0'))
    .replace('%M', String(d.getUTCMinutes()).padStart(2, '0'))
    .replace('%S', String(d.getUTCSeconds()).padStart(2, '0'))
}
```
